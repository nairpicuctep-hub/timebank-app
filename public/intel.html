📌 Add tracking to timebank.academy setup required

Add this one-line script to your main website's <head> to start collecting real visitor data into Supabase. Takes 2 minutes.

<!-- TimeBank Analytics — paste in <head> of timebank.academy -->
<script>
  (function(){
    var d={page:location.pathname,ref:document.referrer,ua:navigator.userAgent,
      w:screen.width,lang:navigator.language,ts:Date.now(),sid:Math.random().toString(36).slice(2)};
    var t=Date.now();
    window.addEventListener('beforeunload',function(){
      d.duration=Math.round((Date.now()-t)/1000);
      navigator.sendBeacon('https://sdtyekfubixjsjmexygx.supabase.co/rest/v1/page_views',
        JSON.stringify(d));
    });
    fetch('https://sdtyekfubixjsjmexygx.supabase.co/rest/v1/page_views',{method:'POST',
      headers:{'Content-Type':'application/json','apikey':'eyJhbGciOiJIUzI1NiIsInR5cCI6IkpXVCJ9.eyJpc3MiOiJzdXBhYmFzZSIsInJlZiI6InNkdHlla2Z1Yml4anNqbWV4eWd4Iiwicm9sZSI6ImFub24iLCJpYXQiOjE3Nzg0OTM1MDQsImV4cCI6MjA5NDA2OTUwNH0.ZczzaVCHAvqBFozKEXmnYQKrot4zPX0WvbbUzqkaVAE'},
      body:JSON.stringify(d)});
  })();
</script>

Also run this SQL in Supabase to create the tracking table:

create table public.page_views (
  id uuid primary key default uuid_generate_v4(),
  created_at timestamptz default now(),
  page text, ref text, ua text, w int,
  lang text, sid text, duration int,
  country text, city text, utm_source text,
  utm_medium text, utm_campaign text
);
alter table public.page_views enable row level security;
create policy "anon insert" on public.page_views for insert using (true);
